<digit> ::= 0-9
<alpha> ::= a-z | A-Z
<ident> ::= (<alpha>|_)(<alpha>|_|<digit>)*

<integer> ::= 0|1-9 <digit>*

Mots clés : 

and def else for if True False
in not or print return None

====================

<file>          ::= NEWLINE? <defs> <stmt><opt_stmts> EOF
<defs>          ::= <def><defs>
                  | ^
<def>           ::= def <ident> (<ident_list>) : <suite>
<suite>         ::= <simple_stmt> NEWLINE
                  | NEWLINE BEGIN <stmt><opt_stmts> END
<simple_stmt>   ::= return <expr>
                  | <ident> = <expr>
                  | <expr> [<expr>] = <expr>
                  | print(<expr>)
                  | <expr>
<opt_stmts>       ::= <stmt><opt_stmts>
                  | ^
<stmt>          ::= <simple_stmt> NEWLINE
                  | if <expr> :
                  | if <expr> : <suite> else : <suite>
                  | for <ident> in <expr> : <suite>
<expr>          ::= <const>
                  | <ident>
                  | <expr>[<expr>]
                  | - <expr>
                  | not <expr>
                  | <expr> <binop> <expr>
                  | <ident> (<expr_list>)
                  | [<expr_list>]
                  | (<expr>)
<binop>         ::= + | - | * | // | % | <= | >= | > | <
                  | != | == | and | or
<const>         ::= <integer> | <string> | True | False | None

# Liste d'expressions séparées par des virgules
<expr_list>         ::= <expr><expr_list_suite>
                      | ^
<expr_list_suite>   ::= ,<expr_list_expr>
                      | ^
<expr_list_expr>    ::= <expr><expr_list_suite>

# Liste d'identificateurs séparées par des virgules
<ident_list>         ::= <ident><ident_list_suite>
                      | ^
<ident_list_suite>   ::= ,<ident_list_ident>
                      | ^
<ident_list_ident>    ::= <ident><ident_list_suite>
